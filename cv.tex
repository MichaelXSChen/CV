\section{Publications}


\begin{itemize}
    \item \textbf{HAMS: High Availability for Distributed Machine Learning Service Graphs.} \\ 
    \authors{\textbf{Xusheng Chen}, Shixiong Zhao, Cheng Wang, Fanxin Li, Ji Qi, Heming Cui, Cheng Li, Sen Wang.}\\
    \conference{Proceedings of the 50th IEEE/IFIP International Conference on Dependable Systems and Networks (DSN '20).}\\
    \authors{\bf* Parallel First Author (annotated in the paper's first page).}\\
    \info{Describes HAMS, an efficient fault-tolerence system for a distributed deployment graph consisting of machine learning serving models (some are stateful) with little performance overhead. HAMS is the first system that can correctly replicate the nondeterminism caused by GPU threads on running AI serving models.} \\

    \item \textbf{UPA: An Automated, Accurate and Efficient Differentially Private Big-data Mining System.} \\ 
    \authors{Tzs On Li, Jianyu Jiang, Ji Qi, Chi Chiu So, Jiacheng Ma, \textbf{Xusheng Chen}, Tianxiang Shen, Heming Cui, Yuexuan Wang, Peng Wang.} \\
    \conference{Proceedings of the 50th IEEE/IFIP International Conference on Dependable Systems and Networks (DSN '20).}\\ 
    \info{Describes UPA, the first automated, efficient and precise differentially private big-data mining system that can preserve individual privacy while supporting general Spark big-data queries.} \\

    \item \textbf{Uranus: Simple, Efficient SGX Programming and Its Applications.} \\
    \authors{Jianyu Jiang, \textbf{Xusheng Chen}, Tzs On Li, Cheng Wang, Tianxiang Shen, Shixiong Zhao, Heming Cui, Cho-Li Wang, Fengwei Zhang.} \\
    \conference{Proceedings of the 15th ACM ASIA Conference on Computer and Communications Security (ASIACCS '20).} \\
    \info{Describes Uranus, the first SGX-compatible and Spark-compatible secure big-data computing system that can practically support typical big-data datasets with our new big-data aware Java memory reuse protocols.} \\

    \item \textbf{Fulva: Efficient Live Migration for In-memory Key-Value Stores with Zero Downtime.} \\
    \authors{Jiewen Hai, Cheng Wang, \textbf{Xusheng Chen}, Tsz On Li, Heming Cui.} \\ 
    \conference{Proceedings of the 38th International Symposium on Reliable Distributed Systems (SRDS '19).} \\
    \info{Describes Fulva, an in-memory key-value store migration system that can almost pertain the store's normal-case performance with no downtime during migration.}

    \item \textbf{PLOVER: Fast, Multi-core Scalable Virtual Machine Fault-tolerance.}\\
    \authors{\textbf{Xusheng Chen}, Cheng Wang, Weiwei Jia, Haoran Qiu, Boxuan Li, Shixiong Zhao, Heming Cui.} \\
    \conference{Proceedings of The 15th USENIX Symposium on Networked Systems Design and Implementation 2018 (NSDI '18).}\\
    \authors{\textbf{*Parallel First Author (annotated in the paper's first page).}} \\
    \info{Describes PLOVER, a multi-core scalable VM fault-tolerance system built on the Virtualized State Machine Replication (VSMR) concept.} \\

    \item \textbf{A Fast, General Storage Replication Protocol for Active-Active Virtual Machine Fault Tolerance.} \\
    \authors{Cheng Wang, \textbf{Xusheng Chen}, Zixu Wang, Youwei Zhu, Heming Cui.} \\
    \conference{Proceedings of the IEEE 23rd International Conference on Parallel and Distributed Systems (ICPADS '18), 2017.}\\
    \info{Describes Gannet, a replication protocol for efficiently replicating virtual machines.} \\

    \item \textbf{Effectively Mitigating I/O Inactivity in vCPU Scheduling.} \\
    \authors{Weiwei Jia, Cheng Wang, \textbf{Xusheng Chen}, Jianchen Shan, Xiaowei Shang, Heming Cui, Xiaoning Ding, Luwei Cheng, F.C.M. Lau, Yuexuan Wang, Yuangang Wang.} \\
    \conference{Proceedings of the 2018 USENIX Annual Technical Conference (ATC '18), 2018.} \\
    \info{Describes vMigrator, a lightweight, easy to use tool that can effectively mitigate I/O inactivity in vCPU scheduling, greatly improving I/O performance in VMs.}\\

    \item \textbf{APUS: Fast and Scalable PAXOS on RDMA.}\\
    \authors{Cheng Wang, Jianyu Jiang, \textbf{Xusheng Chen}, Ning Yi, Heming Cui.}\\
    \conference{Proceedings of the ACM Symposium on Cloud Computing (SOCC '17), 2017.}\\
    \info{Describes APUS, a fast Paxos protocol and its runtime system using fast RDMA features.} \\

    \item \textbf{Eges: Efficient, DoS-resistant Consensus for Permissioned Blockchains.} \\
    \authors{\textbf{Xusheng Chen}, Shixiong Zhao, Cheng Wang, Haoze Song, Jianyu Jiang, Ji Qi, Tsz On Li, T.-H. Hubert Chan, Heming Cui.} \\
    \conference{Submitted to IEEE TDSC, under review.} \\
    \info{Describes Edges, an efficient consensus protocol that can hide consensus nodes to defend against targeted DoS attacks in a large-scale permissioned blockchain system.} \\

    \item \textbf{Chronos: Low-latency, Serializable Transaction Processing in Globally-distributed Edge Data Centers.} \\
    \authors{\textbf{Xusheng Chen}, Haoze Song, Jianyu Jiang, Heming Cui.} \\
    \conference{Submitted, under review.} \\
    \info{Describes Chronos, a fault-tolerant concurrency control protocol that achieves low tail latency and horizontal scalability gloably-distributed database deployed in edge data centers.}

\end{itemize}
